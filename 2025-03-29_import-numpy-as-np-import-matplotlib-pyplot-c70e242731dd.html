import numpy as np
import matplotlib.pyplot as plt

# EFMW-style emergent volatility simulation
np.random.seed(42)

# Parameters
T = 500
cognitive_resonance = 0.75 # Feedback intensity
recursive_pressure = 0.35 # Influence from past gradients
baseline_entropy = 0.02
phase_shift = 1.5 * np.pi # Phase modulation
noise = np.random.normal(0, 1, T)

# State variables
force = np.zeros(T)
velocity = np.zeros(T)
acceleration = np.zeros(T)
coherence = np.zeros(T)

# Initialize
force[0] = 0.1

for t in range(1, T):
 # Emergent coherence based on recursive resonance
 coherence[t] = np.sin(phase_shift * t / T) * np.exp(-t / T)
 
 # Recursive acceleration driven by force-space memory
 acceleration[t] = (force[t-1] * cognitive_resonance + 
 velocity[t-1] * recursive_pressure + 
 coherence[t] + baseline_entropy * noise[t])
 
 velocity[t] = velocity[t-1] + acceleration[t]
 force[t] = force[t-1] + velocity[t]

# Plot results
plt.figure(figsize=(10, 5))
plt.plot(force, label='Force-space Signal')
plt.plot(velocity, label='Velocity (Impulse)')
plt.plot(coherence, label='Coherence Phase')
plt.title('EFMW Emergent Signal Simulation')
plt.legend()
plt.grid(True)
plt.show()
